-schema=smtschm.c

After-Init:
    (--) Ok                                 -> Before-Opening
          +
    (--) Error                              ->
          + Terminate-The-Thread

Before-Opening:
    (--) Open                               -> Autfile-Open
          + Open-Authorisation-File
    (--) Exception                          -> Autfile-Missing
          +
    (--) Check-Url                          -> Before-Opening
          + Reply-Not-Protected
    (--) Check-Name                         -> Before-Opening
          + Reply-Not-Protected
    (--) Close                              ->
          + Terminate-The-Thread

Autfile-Open:
    (--) Reopen                             -> Autfile-Open
          + Close-Authorisation-File
          + Open-Authorisation-File
    (--) Check-Url                          -> Autfile-Open
          + Check-If-Url-Protected
    (--) Check-Name                         -> Autfile-Open
          + Check-User-Name
    (--) Close                              ->
          + Close-Authorisation-File
          + Terminate-The-Thread

Autfile-Missing:
    (--) Reopen                             -> Autfile-Missing
          +
    (--) Check-Url                          -> Autfile-Missing
          + Reply-Not-Protected
    (--) Check-Name                         -> Autfile-Missing
          + Reply-Not-Protected
    (--) Close                              ->
          + Terminate-The-Thread

Defaults:
    (--) Exception                          ->
          + Close-Authorisation-File
          + Terminate-The-Thread
    (--) Shutdown                           ->
          + Close-Authorisation-File
          + Terminate-The-Thread
    (--) $other                             ->
          +
